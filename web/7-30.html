		//登记按钮
		function quedin(b){
			var row = $('#dg').datagrid('getSelected');
			if (row) {
				$('#dlg').dialog('open').dialog('setTitle','登记');
				$('#cc').combobox({
				    url:'xpgz_combobox_json.asp',
				    valueField:'id',
				    textField:'text'
				});
			}else {
				tsshow('请选择一条数据！');
			}
		}

		//确定按钮
		function savetype() {
			var row = $('#dg').datagrid('getSelected');
			$.post('xpgz_json_edit.asp',{ id:row.GZ_SPDM,va:$('#roleID').textbox('getValue') },function(res){ 
				$('#dlg').dialog('close');
				$('#dg').datagrid('reload');
				$('#fm').form('clear');
			},'json');
		}

		//查找按钮
		function doSearch()
		{
			$('#dg').datagrid('load',{
				mc: $("#mc").textbox('getValue')
			});
		}
